pcminStart

int pcminStart(unsigned long rate, unsigned long channels, float *buffer, unsigned long bufferlen)

unsigned long rate [in]
desired sample rate, in samples per second per channel

unsigned long channels [in]
desired number of channels

float *buffer [in]
pointer to a circular buffer that must be at least [channels * bufferlen * sizeof(float)] bytes large

unsigned long bufferlen [in]
size of buffer in blocks (channels * sizeof(float))

int [return]
0 = success
1 = unsupported format
